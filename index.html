metar_data = {}

metar_data['report'] = ds.report.values.astype(str)

metar_data['lon'] = ds.longitude.values[ds.stationIndex.values] * units.deg
metar_data['lat'] = ds.latitude.values[ds.stationIndex.values] * units.deg
metar_data['elev'] = ds.stationAltitude.values[ds.stationIndex.values] * units.meter
metar_data['time'] = ds.time.values.astype('datetime64[m]')
metar_data['stid'] = ds.station_id[ds.stationIndex.values].values.astype(str)
metar_data['stn_desc'] = ds.station_description[ds.stationIndex.values].values.astype(str)
metar_data['wdir'] = ds.wind_from_direction.values * units.degree
metar_data['wspd'] = ds.wind_speed.values * units.meter_per_second
metar_data['temp'] = ds.air_temperature.values * units.degC
metar_data['dewp'] = ds.dew_point_temperature.values * units.degC
metar_data['pres'] = ds.air_pressure_at_sea_level.values * units.hectopascal
metar_data['wx'] = ds.weather.values.astype(str)
metar_data['wx'] = wx_code_to_numeric(metar_data['wx'])
metar_data['altim'] = ds.hectoPascal_ALTIM.values * units.hectopascal
metar_data['altim'] = altimeter_to_sea_level_pressure(metar_data['altim'], metar_data['elev'], metar_data['temp'])
metar_data['cloud_cover'] = np.nan_to_num(ds.cloud_area_fraction.values, copy=False)
metar_data['cloud_cover'] = (8 * metar_data['cloud_cover']).astype(int)

def station_to_slp(p, h, t):
    return p.to(units.hectopascal) * (1 - (0.0065*h.to(units.meter).m)/(t.to(units.kelvin).m+0.0065*h.to(units.meter).m)) ** -5.257
